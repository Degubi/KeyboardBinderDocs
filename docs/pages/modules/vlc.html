VLC module

Contains functions used to interact with VLC


Importing

from keyboardBinder import VLC

Functions

add_item_to_playlist(file_path: str) -> None

Description:

Adds the given file to the playlist

Example:

VLC.add_item_to_playlist(file_path = 'C:/Users/User/Desktop/In Flames/Subterranean/Stand Ablaze.mp3')

get_player_volume() -> int

Description:

Returns the player's volume in percentage

Example:

volume_level = VLC.get_player_volume()

get_playlist_items() -> list[str]

Description:

Returns the current playlist's items by file path

Example:

item_paths = VLC.get_playlist_items()

is_player_paused() -> bool

Description:

Returns true if the player is in a paused state

Example:

is_player_paused = VLC.is_player_paused()

is_player_playing() -> bool

Description:

Returns true if the player is in a playing state

Example:

is_player_playing = VLC.is_player_playing()

is_player_stopped() -> bool

Description:

Returns true if the player is in a stopped state

Example:

is_player_stopped = VLC.is_player_stopped()

jump_to_next_playlist_item() -> None

Description:

Jumps to the next item in the current playlist

Example:

VLC.jump_to_next_playlist_item()

jump_to_previous_playlist_item() -> None

Description:

Jumps to the previous item in the current playlist

Example:

VLC.jump_to_previous_playlist_item()

pause_player() -> None

Description:

Pauses the player

Example:

VLC.pause_player()

set_player_volume(volume_level: int) -> None

Description:

Sets the player's volume to the given percentage value

Example:

VLC.set_player_volume(volume_level = 25)

start_player() -> None

Description:

Starts the player

Example:

VLC.start_player()

stop_player() -> None

Description:

Stops the player

Example:

VLC.stop_player()
