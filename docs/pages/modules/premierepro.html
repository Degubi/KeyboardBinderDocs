PremierePro module

Contains functions used to interact with Premiere Pro


Importing

from keyboardBinder import PremierePro

Functions

add_audio_effect_to_selection(effect_name: str, properties: dict) -> None

Description:

Applies the given audio effect by name to all the selected audio clips

Example:

PremierePro.add_audio_effect_to_selection(effect_name = 'Unknown', properties = '{}')

add_video_effect_to_selection(effect_name: 'Sharpen', properties: { amount: int }) -> None

Description:

Applies the Sharpen video effect to all the selected video clips

Example:

PremierePro.add_video_effect_to_selection(effect_name = 'Sharpen', properties = '{}')

add_video_effect_to_selection(effect_name: str, properties: dict) -> None

Description:

Applies the given video effect by name to all the selected video clips

Example:

PremierePro.add_video_effect_to_selection(effect_name = 'Unknown', properties = '{}')

adjust_audio_gain_level_on_selection(level: float) -> None

Description:

Adjusts the gain by the specified amount on the selected audio clips

Example:

PremierePro.adjust_audio_gain_level_on_selection(level = 5)

apply_audio_preset_to_selection(preset_path: str) -> None

Description:

Applies the given preset to the selected audio clips. Note: The preset must be imported from a preset file. See the 'import_presets_from_file' function

Example:

PremierePro.apply_audio_preset_to_selection(preset_path = 'Presets/CoolAudio')

apply_video_preset_to_selection(preset_path: str) -> None

Description:

Applies the given preset to the selected video clips. Note: The preset must be imported from a preset file. See the 'import_presets_from_file' function

Example:

PremierePro.apply_video_preset_to_selection(preset_path = 'Presets/CoolVideo')

get_audio_gain_level_from_selection() -> float

Description:

Returns the audio gain value from the selected clip

Example:

gain_level = PremierePro.get_audio_gain_level_from_selection()

get_media_paths_from_selection() -> list[str]

Description:

Returns the media paths of the selected clips. Note: The list might contain duplicate paths according to the current selection

Example:

media_paths = PremierePro.get_media_paths_from_selection()

get_scale_to_frame_size_from_selection() -> bool

Description:

Returns the scaleToFrameSize property's value from the selected clip

Example:

scale_enabled = PremierePro.get_scale_to_frame_size_from_selection()

get_speed_from_selection() -> int

Description:

Returns the speed percentage value from the selected clip

Example:

speed = PremierePro.get_speed_from_selection()

insert_item_at_player_position(item_path: str) -> None

Description:

Inserts a clip/sequence at the given path. Root is the project path, bins are separated by '/' characters

Example:

PremierePro.insert_item_at_player_position(item_path = 'Bin1/Bin2/ClipOrSequenceName')

is_track_targeted(track_name: str) -> bool

Description:

Returns whether the given track is targeted. If the track doesn't exist False gets returned.

Example:

is_targeted = PremierePro.is_track_targeted(track_name = 'Video 1')

remove_effects_from_selection() -> None

Description:

Removes all effects from the selected clips

Example:

PremierePro.remove_effects_from_selection()

set_audio_gain_level_on_selection(level: float) -> None

Description:

Sets the gain to the specified amount on the selected audio clips

Example:

PremierePro.set_audio_gain_level_on_selection(level = 5)

set_scale_to_frame_size_on_selection(enabled: bool) -> None

Description:

Sets the scaleToFrameSize property to the given value on the selected video clips

Example:

PremierePro.set_scale_to_frame_size_on_selection(enabled = True)

set_speed_on_selection(value: int) -> None

Description:

Sets the speed percentage value to the given value on the selected clip

Example:

PremierePro.set_speed_on_selection(value = 42)

set_track_targeted(track_name: str, targeted: bool) -> None

Description:

Sets the given track's targeting state to the given value

Example:

PremierePro.set_track_targeted(track_name = 'Video 1', targeted = True)
