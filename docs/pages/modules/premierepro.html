PremierePro module

Contains functions used to interact with Premiere Pro


Importing

from keyboardBinder import PremierePro

Functions

add_audio_effect_to_selection(effect_name: str) -> None

Description:

Applies the given audio effect by name to all the selected audio clips

Example:

PremierePro.add_audio_effect_to_selection(effect_name = 'Bass')

add_video_effect_to_selection(effect_name: str) -> None

Description:

Applies the given video effect by name to all the selected video clips

Example:

PremierePro.add_video_effect_to_selection(effect_name = 'Gamma Correction')

adjust_audio_gain_level_on_selection(level: float) -> None

Description:

Adjusts the gain by the specified amount on the selected audio clips

Example:

PremierePro.adjust_audio_gain_level_on_selection(level = 5)

get_audio_gain_level_from_selection() -> float

Description:

Returns the audio gain value from the selected clip

Example:

gain_level = PremierePro.get_audio_gain_level_from_selection()

get_scale_to_frame_size_from_selection() -> bool

Description:

Returns the scaleToFrameSize property's value from the selected clip

Example:

scale_enabled = PremierePro.get_scale_to_frame_size_from_selection()

get_selected_clip_media_paths() -> list[str]

Description:

Returns the media paths of the selected clips. Note: The list might contain duplicate paths according to the current selection

Example:

media_paths = PremierePro.get_selected_clip_media_paths()

insert_item_at_player_position(item_path: str) -> None

Description:

Inserts a clip/sequence at the given path. Root is the project path, bins are separated by '/' characters

Example:

PremierePro.insert_item_at_player_position(item_path = 'Bin1/Bin2/ClipOrSequenceName')

list_sequence_names() -> list[str]

Description:

Returns the list of sequence names in the project

Example:

sequence_names = PremierePro.list_sequence_names()

remove_effects_from_selection() -> None

Description:

Removes all effects from the selected clips

Example:

PremierePro.remove_effects_from_selection()

set_audio_gain_level_on_selection(level: float) -> None

Description:

Sets the gain to the specified amount on the selected audio clips

Example:

PremierePro.set_audio_gain_level_on_selection(level = 5)

set_scale_to_frame_size_on_selection(enabled: bool) -> None

Description:

Sets the scaleToFrameSize property to the given value on the selected video clips

Example:

PremierePro.set_scale_to_frame_size_on_selection(enabled = True)
