OBS

restartMediaSource(source: string)

Description:

Restarts the specified media source

Example:

OBS.restartMediaSource(source = 'hidden-porn')

setMediaSourcePaused(source: string, paused: boolean)

Description:

Plays or pauses the give media source based on the given value

Example:

OBS.setMediaSourcePaused(source = 'epic-frag', paused = true)

setSourceMuted(source: string, muted: boolean)

Description:

Sets the specified media source's mute state to the given value

Example:

OBS.setSourceMuted(source = 'duc-duc', muted = false)

startRecording()

Description:

Starts recording

Example:

OBS.startRecording()

stopMediaSource(source: string)

Description:

Stops the given media source

Example:

OBS.stopMediaSource(source = 'background-music')

toggleMediaSourcePause(source: string)

Description:

Plays or pauses the given media source based on whether it is currently playing or not (toggling it)

Example:

OBS.toggleMediaSourcePause(source = 'intro')

toggleSourceMute(source: string)

Description:

Toggles the specified meda source's mute state

Example:

OBS.toggleSourceMute(source = 'outro-music')
